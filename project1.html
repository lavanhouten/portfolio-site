# counts the number of crimes that occurred in each neighborhood
crime_counts <- crimes_p %>%
  filter(year==2021) %>%
  group_by(neighborho) %>%
  summarize(count = n())

# drops coordinates in order to join
crime_counts <- st_drop_geometry(crime_counts)

# joins count of crimes per neighborhood to the rest of the neighborhood stats
neighborhoods_p <- neighborhoods %>%
  left_join(crime_counts, by = c("nhood_name" = "neighborho"))

# deals with na values
neighborhoods_p$count <- ifelse(is.na(neighborhoods_p$count), 0, neighborhoods_p$count)

# adds the color that each neighborhood should appear on the map
neighborhoods_p$color <- ifelse(neighborhoods_p$count > 1200,"red", ifelse(neighborhoods_p$count < 100, "green", "grey"))

# creates neighborhood map
tm_shape(neighborhoods_p) +
  tm_polygons(alpha = 0.7) +
  tm_shape(neighborhoods_p) +
  tm_fill(col = "color") +
  tm_borders(lwd = 0) +
  tm_layout(main.title = "Crime in Detroit by Neighborhood", frame= FALSE) +
  tm_legend()

# counts the number of crimes in each zipcode
zip_counts <- crimes_p %>%
  filter(year==2021) %>%
  group_by(zip_code) %>%
  summarize(count = n())

# drops coordinates in order to join
zip_counts <- st_drop_geometry(zip_counts)

# adds crime count per zip codes 
zipcodes_p <- zipcodes_p %>%
  left_join(zip_counts, by = c("zipcode" = "zip_code"))

# accounts for na values
zipcodes_p$count <- ifelse(is.na(zipcodes_p$count), 0, zipcodes_p$count)

# adds the color to the zip code statistics 
zipcodes_p$color <- ifelse(zipcodes_p$count > 5000,"red", ifelse(zipcodes_p$count < 1000, "green", "grey"))

# creates zip code map
tm_shape(zipcodes_p) +
  tm_polygons(alpha = 0.7) +
  tm_shape(zipcodes_p) +
  tm_fill(col = "color") +
  tm_borders(lwd = 0) +
  tm_layout(main.title = "Crime in Detroit by Zipcode", frame= FALSE) +
  tm_text("zipcode", size=0.6)

#  selects only robberies that occurred in 2021
filter_counts <- crimes_p %>%
  filter(year==2021, charge_des =="ROBBERY")

# creates 0.5 mile and 1 mile buffers around grocery stores
gro_half_buff <- st_buffer(grocery_p, dist = 2640)
gro_1m_buff <- st_buffer(grocery_p, dist = 5280)

# mapped out these buffers
tm_shape(neighborhoods_p) + tm_polygons(alpha = 0.4) +
  tm_shape(grocery_p) + tm_dots(col = "blue", size = 0.1) +
  tm_shape(gro_half_buff) + tm_borders(col = "red") + tm_fill(col = "red", alpha = 0.6) +
  tm_shape(gro_1m_buff) + tm_borders(col = 'pink') + tm_fill(col = 'pink', alpha = 0.5) +
  tm_layout(main.title = "Detroit Grocery Stores", frame= FALSE)

# found which robberies occurred close to which grocery stores
rob_within_half <- st_within(filter_counts, gro_half_buff, sparse = FALSE)
rob_within_1m <- st_within(filter_counts, gro_1m_buff, sparse = FALSE)

# dropped the geometry to prepare to export
rob_within_half <- st_drop_geometry(rob_within_half)
rob_within_1m <- st_drop_geometry(rob_within_1m)

# counted the number of robberies that occurred within 0.5 and 1 mile of grocery stores
robs_within_half <- apply(X = rob_within_half, MARGIN=2,FUN=sum)
robs_within_1m <- apply(X = rob_within_1m, MARGIN=2,FUN=sum)

# combined the rest of the grocery store data with the number of robberies close to a given store
grocery_robs <- cbind(grocery, robs_within_half, robs_within_1m)

# selected the store name, address, robberies within 0.5 mile, and robberies within 1 mile for export
grocery_robs1 <- grocery_robs %>% 
  select(Store_Name, Address, robs_within_half, robs_within_1m)
